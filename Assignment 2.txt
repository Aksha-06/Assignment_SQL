1)
SELECT DEPTNO, DNAME
FROM DEPT D
WHERE DEPTNO NOT IN
                   (SELECT DEPTNO
                    FROM EMP);

2)
SELECT ENAME,JOB
FROM EMP
WHERE MGR IS NULL;

3)
SELECT E.ENAME, E.JOB, D.LOC, E.COMM, D.DNAME
FROM EMP E, DEPT D
WHERE D.LOC='NEW YORK' AND E.COMM>1000;

4)
SELECT COUNT(ENAME) AS 'NO. OF EMP IN CHICAGO'
FROM EMP
WHERE DEPTNO = 
              (SELECT DEPTNO 
               FROM DEPT 
               WHERE LOC = 'CHICAGO');

5)
SELECT ENAME, DEPTNO, JOB
FROM EMP
WHERE DEPTNO = 
              (SELECT DEPTNO 
               FROM DEPT 
               WHERE LOC = 'CHICAGO');

6)
SELECT E.ENAME, D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
ORDER BY LOC;

7)
SELECT COUNT(DEPTNO) AS 'DEPT AND EMP COUNT'
FROM DEPT
UNION
SELECT COUNT(ENAME)
FROM EMP;

8)
SELECT *
FROM EMP E, EMP M
WHERE E.MGR = M.EMPNO AND E.HIREDATE < M.HIREDATE;

9)
SELECT COUNT(JOB) AS 'MANAGER COUNT'
FROM EMP
WHERE  JOB = 'MANAGER';

10)
SELECT ENAME, JOB
FROM EMP
WHERE MGR IS NULL;

11)
SELECT ENAME, DEPTNO, JOB
FROM EMP
WHERE ENAME = 'SMITH';

12)
SELECT ENAME, DEPTNO, JOB
FROM EMP
WHERE ENAME = 'SMITH';

13)
SELECT E.*, D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO;

14)
SELECT ENAME, SAL, JOB
FROM EMP
WHERE JOB = 'SALESMAN' AND
                            SAL <
                                 (SELECT SAL
                                  FROM EMP
                                  WHERE ENAME= 'MILLER');

15)
SELECT E.*, D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO AND D.LOC = 'CHICAGO';

16)
SELECT ENAME, SAL
FROM EMP
WHERE SAL 
          >
            (SELECT MIN(SAL)
             FROM EMP
             WHERE JOB= 'MANAGER');

17)
SELECT E.ENAME, E.EMPNO, E.DEPTNO, E.JOB, D.DNAME
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO AND D.DNAME = 'ACCOUNTING';

18)
SELECT EMPNO,ENAME, JOB
FROM EMP
WHERE JOB = 'MANAGER';

19)
SELECT DNAME,
CASE
    WHEN DNAME = 'ACCOUNTING' THEN 'NOT SALES'
    WHEN DNAME = 'RESEARCH' THEN 'NOT SALES'
    WHEN DNAME = 'OPERATIONS' THEN 'NOT SALES'
    ELSE 'SALES'
    END AS 'DESCRIPTION'
FROM DEPT;

20)
SELECT ENAME, SAL,
CASE
    WHEN SAL > 1000 THEN SAL
    ELSE '1000'
    END AS 'DESCRIPTION'
FROM EMP;

21)
SELECT ENAME, SAL,
CASE 
    WHEN SAL < 1000 THEN 'LOW'
    WHEN SAL BETWEEN 1000 AND 2500 THEN 'MEDIUM'
    WHEN SAL > 2500 THEN 'HIGH'
    END AS 'SAL RANGE'
FROM EMP;
